\begin{frame}{Vì sao không chỉ dùng nén ngữ cảnh?}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Nén ngữ cảnh}
        \begin{itemize}
          \item Nén văn bản dài thành biểu diễn ẩn ngắn hơn.
          \item Hữu ích cho suy luận với ngữ cảnh dài.
          \item Không tối ưu riêng cho tóm tắt bám nguồn.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{LLM2Seq}
        \begin{itemize}
          \item Giữ token-level memory cho từng token nguồn.
          \item Decoder đọc nguồn bằng cross-attention.
          \item Giảm nguy cơ mất thực thể, số liệu, bước thao tác.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

\begin{frame}{Adapter: biến hidden states thành memory}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.53\textwidth}
      \begin{block}{Ý tưởng}
        \begin{itemize}
          \item Lấy hidden states từ source encoder.
          \item Kết hợp nhiều tầng ẩn để giữ cả thông tin nông và sâu.
          \item Gated residual projection đưa biểu diễn sang không gian decoder.
          \item Decoder đọc source memory bằng cross-attention.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.43\textwidth}
      \begin{center}
      \begin{tikzpicture}[node distance=0.5cm, every node/.style={font=\small}]
        \node[draw,rounded corners,fill=LLMBlue,minimum width=3.2cm,minimum height=0.7cm] (enc) {Hidden states};
        \node[draw,rounded corners,fill=LLMGreen,below=of enc,minimum width=3.2cm,minimum height=0.7cm] (fuse) {Kết hợp tầng ẩn};
        \node[draw,rounded corners,fill=LLMOrange,below=of fuse,minimum width=3.2cm,minimum height=0.7cm] (gate) {Gated adapter};
        \node[draw,rounded corners,fill=LLMBlue,below=of gate,minimum width=3.2cm,minimum height=0.7cm] (mem) {Source memory};
        \draw[-{Latex[length=2mm]},thick] (enc) -- (fuse);
        \draw[-{Latex[length=2mm]},thick] (fuse) -- (gate);
        \draw[-{Latex[length=2mm]},thick] (gate) -- (mem);
      \end{tikzpicture}
      \end{center}
    \end{column}
  \end{columns}

  \vfill
  \centering
  \normalsize\alert{Mục tiêu: tái sử dụng biểu diễn mạnh mà vẫn giữ ranh giới encoder-decoder.}

  \smallskip
  \small Adapter là lớp nối quan trọng: không tạo encoder mới từ đầu, mà chuyển hidden states thành memory dùng được cho decoder.
\end{frame}

\begin{frame}{Self-distilled MTP: tăng tốc có kiểm chứng}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Huấn luyện}
        \begin{itemize}
          \item Huấn luyện mô hình tóm tắt chính trước.
          \item Đóng băng encoder, adapter và decoder.
          \item Chỉ huấn luyện 4 MTP heads bằng CE + KL.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Suy luận}
        \begin{itemize}
          \item MTP heads dự đoán trước các token tương lai.
          \item Decoder chính kiểm chứng từng token nháp.
          \item Chỉ sinh ra token đã được kiểm chứng.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vfill
  \begin{center}
    \begin{tikzpicture}[node distance=0.7cm, every node/.style={font=\small}]
      \node[draw,rounded corners,fill=LLMBlue,minimum width=2.4cm,minimum height=0.72cm] (main) {Decoder chính};
      \node[draw,rounded corners,fill=LLMOrange,right=1.0cm of main,minimum width=2.5cm,minimum height=0.72cm] (draft) {Token nháp};
      \node[draw,rounded corners,fill=LLMGreen,right=1.0cm of draft,minimum width=2.5cm,minimum height=0.72cm] (verify) {Kiểm chứng};
      \node[draw,rounded corners,fill=LLMBlue,right=1.0cm of verify,minimum width=2.2cm,minimum height=0.72cm] (emit) {Sinh ra};

      \node[below=0.25cm of verify,color=LLMRed] {chỉ nhận token đã được kiểm chứng};
    \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}{Độ trễ MTP bị nghẽn ở đâu?}

      \end{block}

  \end{columns}

\begin{frame}{Thông điệp chính}
  \vspace{0.3em}
  \begin{itemize}
    \item LLM2Seq dùng LLM source encoder để tạo memory cho decoder.
    \item Qwen cho chất lượng tốt hơn Llama và độ dài gần tham chiếu hơn.
    \item Verified MTP đạt 1.17x, nhưng bị giới hạn bởi bước kiểm chứng.
  \end{itemize}

  \end{block}
